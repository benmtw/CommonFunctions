153def is_personalized_email(email: str) -> bool:
154    """Heuristically detect whether an email looks person-specific.
155
156    The check is intentionally simple and fast. It rejects obvious role-based
157    local parts such as ``info`` and ``support``, as well as numeric-only and
158    malformed values.
159
160    Args:
161        email: Email address to classify.
162
163    Returns:
164        ``True`` when the local part looks person-like, else ``False``.
165    """
166    parsed = _split_email(email)
167    if parsed is None:
168        return False
169
170    local, _domain = parsed
171    base_local = local.split("+", 1)[0]
172    if not base_local:
173        return False
174
175    if base_local in _ROLE_BASED_LOCAL_PARTS:
176        return False
177
178    for role in _ROLE_BASED_LOCAL_PARTS:
179        if (
180            base_local.startswith(f"{role}.")
181            or base_local.startswith(f"{role}_")
182            or base_local.startswith(f"{role}-")
183        ):
184            return False
185
186    if base_local.isdigit():
187        return False
188    if len(base_local) < 2:
189        return False
190    if re.search(r"[a-z]", base_local) is None:
191        return False
192    return True

196    """Return whether a domain/email belongs to a known free provider.

201    Args:

204
205    Returns:
206        ``True`` if the domain is in the free-provider dataset, else ``False``.
207    """
208    parsed = _split_email(domain_or_email)
209    if parsed is not None:
210        _local, domain = parsed
211        return _domain_in_set(domain, get_free_provider_domains())
212
213    domain = _normalize_domain(domain_or_email)
214    if domain is None:
215        return False
216    return _domain_in_set(domain, get_free_provider_domains())

 86        """Fetch and decode a KV value.
 87
 88        Args:
 89            key: Exact KV key.
 90
 91        Returns:
 92            Parsed JSON value as a dictionary, or ``None`` when the key does
 93            not exist or has empty content.
 94        """
 95        req = request.Request(
 96            self._key_url(key),
 97            method="GET",
 98            headers={
 99                "Authorization": f"Bearer {self._config.api_token}",
100            },
101        )
102        try:
103            with request.urlopen(req, timeout=self._timeout_seconds) as resp:
104                payload = resp.read().decode("utf-8")
105        except error.HTTPError as exc:
106            if exc.code == 404:
107                return None
108            raise
109        if not payload:
110            return None
111        return json.loads(payload)

128    def _query_url(self) -> str:
129        return (
130            "https://api.cloudflare.com/client/v4/accounts/"

134    def _execute(self, sql: str, params: list[Any] | None = None) -> list[dict[str, Any]]:
135        body = json.dumps(
136            {
137                "sql": sql,
138                "params": params or [],
139            },
140            separators=(",", ":"),
141        ).encode("utf-8")
142        req = request.Request(
143            self._query_url(),
144            method="POST",
145            data=body,
146            headers={
147                "Authorization": f"Bearer {self._config.api_token}",
148                "Content-Type": "application/json",
149            },
150        )
151        with request.urlopen(req, timeout=self._timeout_seconds) as resp:
152            payload = json.loads(resp.read().decode("utf-8"))
153
154        result = payload.get("result")
155        if not isinstance(result, list) or not result:
156            return []
157        first = result[0]
158        if not isinstance(first, dict):
159            return []
160        rows = first.get("results")
161        if isinstance(rows, list):
162            return [row for row in rows if isinstance(row, dict)]
163        return []

165    def get_domain_rating(self, domain: str) -> dict[str, Any] | None:

171            "naming_format_distribution_json, raw_naming_format_codes_json, aggregated_at, source "
172            f"FROM {D1_DOMAIN_RATINGS_TABLE} WHERE domain = ? LIMIT 1"
