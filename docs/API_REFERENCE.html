115def is_personalized_email(email: str) -> bool:
116    """Return True when the local-part appears to represent a person."""
117    parsed = _split_email(email)
118    if parsed is None:
119        return False
120
121    local, _domain = parsed
122    base_local = local.split("+", 1)[0]
123    if not base_local:
124        return False
125
126    if base_local in _ROLE_BASED_LOCAL_PARTS:
127        return False
128
129    for role in _ROLE_BASED_LOCAL_PARTS:
130        if (
131            base_local.startswith(f"{role}.")
132            or base_local.startswith(f"{role}_")
133            or base_local.startswith(f"{role}-")
134        ):
135            return False
136
137    if base_local.isdigit():
138        return False
139    if len(base_local) < 2:
140        return False
141    if re.search(r"[a-z]", base_local) is None:
142        return False
143    return True

51    def _key_url(self, key: str) -> str:
52        encoded_key = parse.quote(key, safe="")
53        return (
54            "https://api.cloudflare.com/client/v4/accounts/"
55            f"{self._config.account_id}/storage/kv/namespaces/"
56            f"{self._config.namespace_id}/values/{encoded_key}"
57        )
58
59    def get(self, key: str) -> dict[str, Any] | None:
60        req = request.Request(
61            self._key_url(key),
62            method="GET",
63            headers={
64                "Authorization": f"Bearer {self._config.api_token}",
65            },
66        )
67        try:
68            with request.urlopen(req, timeout=self._timeout_seconds) as resp:
69                payload = resp.read().decode("utf-8")
70        except error.HTTPError as exc:
71            if exc.code == 404:
72                return None
73            raise
74        if not payload:
75            return None
76        return json.loads(payload)
77
78    def set(self, key: str, value: dict[str, Any]) -> None:
79        body = json.dumps(value, separators=(",", ":")).encode("utf-8")
80        req = request.Request(
81            self._key_url(key),
82            method="PUT",
83            data=body,
84            headers={
85                "Authorization": f"Bearer {self._config.api_token}",
86                "Content-Type": "application/json",
87            },
88        )
89        with request.urlopen(req, timeout=self._timeout_seconds):
90            return

def get_domain_search_cached( *, domain: str, hunter_client: HunterClient, cache_store: common_functions.hunter.CacheStore | None = None, ttl_hours: int = 720) -> dict[str, typing.Any]: 
 94def get_domain_search_cached(
 95    *,
 96    domain: str,
 97    hunter_client: HunterClient,
 98    cache_store: CacheStore | None = None,
 99    ttl_hours: int = 24 * 30,
100) -> dict[str, Any]:
101    """Cache-first wrapper for Hunter domain search results."""
102    normalized = domain.strip().lower()
103    cache_key = f"hunter:domain-search:{normalized}"
104    now = datetime.now(tz=timezone.utc)
105
106    if cache_store is not None:
107        cached = cache_store.get(cache_key)
108        if cached and not _is_stale(cached, now):
109            return cached["result"]
110
111    result = hunter_client.domain_search(normalized)
112    expires = now + timedelta(hours=ttl_hours)
113    to_store = {
114        "source": "hunter.io",
115        "retrieved_at": now.isoformat().replace("+00:00", "Z"),
116        "expires_at": expires.isoformat().replace("+00:00", "Z"),
117        "result": result,
118    }
119    if cache_store is not None:
120        cache_store.set(cache_key, to_store)
121    return result
